86 static int** init_matrix(const std::string& data, int width, int height) {
87  using namespace std;
88 
89  assert(width > 0);
90  assert(height > 0);
91 
92  // alloc()a memoria para a matriz
93  int** img = new int*[height];
94  for (int i = 0; i < height; ++i) {
95  img[i] = new int[width];
96  for (int j = 0; j < width; ++j)
97  img[i][j] = 0;
98  }
99 
100  int i = 0, j = 0;
101  for (const auto& c : data) {
102  // ignorar whitespace
103  if (isspace(c))
104  continue;
105 
106  // preenche a matriz
107  img[i][j] = c - '0';
108 
109  // confere se chegou ao fim de uma linha
110  if (++j >= width) {
111  j = 0;
112  // sai do loop se chegou na altura desejada
113  if (++i >= height)
114  break;
115  }
116  }
117 
118  return img;
119 }
